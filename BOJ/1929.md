```python
import sys, math
input = sys.stdin.readline
M, N = map(int, input().strip().split())

def is_prime(x):
    if x == 0 or x == 1:
        return False
    
    for i in range(2, int(math.sqrt(x))+1):
        if x % i == 0:
            return False
    return True

for n in range(M, N+1):
    if is_prime(n):
        print(n)
```
