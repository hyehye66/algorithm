# N과 M (2)

```python
# 조합
import sys
sys.stdin = open('input.txt', 'r')
input = sys.stdin.readline
N, M = map(int, input().strip().split())
arr = []

def dfs(start):
    if len(arr) == M:
        print(' '.join(map(str, arr)))
        return

    # arr의 요소 정렬이 오름차순이 되기 위해(더 작은 숫자가 추후에 추가되는 것을 막기 위해)
    # start부터 N까지 범위 설정
    for i in range(start, N+1):
        if i in arr:
            continue
        arr.append(i)
        dfs(i+1)
        arr.pop()

dfs(1)
```
