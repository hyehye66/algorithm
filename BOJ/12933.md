import sys
input = sys.stdin.readline
s = list(input().strip())
ans = 0

if len(s)%5 != 0:
    print(-1)
    exit(0)

for k in range(len(s)//5, 0, -1):
    i, j = 0, 0
    temp = 0
    duck_lst = []

    while i < len(s):
        if s[i] == 'quack'[j]:
            duck_lst.append(i)
            if j == 4:
                for l in range(5):
                    s.pop(duck_lst.pop(0)-l)
                i -= 5
                temp += 1

            j = (j+1)%5

        i += 1
    if temp > 0:
        ans += 1

if ans == 0 or s:
    print(-1)
else:
    print(ans)
