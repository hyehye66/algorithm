   ```python
   # ex. 55-50+40
   exp = input().split('-')  # ['55', '50+40']
   num = []
   for i in exp:
       cnt = 0
       sum = i.split('+')  # i='55': sum = ['55'], i='50+40': sum = ['50', '40']
       for j in sum:
           cnt += int(j)
       num.append(cnt)
   ans = num[0]  # 55
   for i in range(1, len(num)):
       ans -= num[i]
   print(ans)
   ```
