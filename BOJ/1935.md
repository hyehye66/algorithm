```python
import sys
input = sys.stdin.readline
N = int(input().strip())
formula = input().strip()
value_lst = list(int(input().strip()) for _ in range(N))
alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
numbers = []

for i in range(len(formula)):
    if formula[i] in alphabet:
        idx = alphabet.index(formula[i])
        numbers.append(value_lst[idx])
    else:
        x = numbers.pop(-2)
        y = numbers.pop(-1)
        if formula[i] == '+':
            numbers.append(x + y)
        elif formula[i] == '-':
            numbers.append(x - y)
        elif formula[i] == '*':
            numbers.append(x * y)
        elif formula[i] == '/':
            numbers.append(x / y)
            
print(f'{numbers[0]:.2f}')  # f'{ì‹¤ìˆ˜:.í‘œê¸°í•  ìë¦¬ìˆ˜f}'
```
