import sys, heapq
input = sys.stdin.readline
N = int(input().strip())
heap = []
d = dict()
for i in range(N):
    x = int(input().strip())
    if x != 0:
        heapq.heappush(heap, abs(x))
        if abs(x) in d:
            d[abs(x)] += [x]
        else:
            d[abs(x)] = [x]

    else:
        if heap:
            abs_val = heapq.heappop(heap)
            min_val = min(d.get(abs_val))
            print(min_val)
            d.get(abs_val).remove(min_val)

        else:
            print(0)
