# N과 M (1)

```python
# 순열
import sys
input = sys.stdin.readline
N, M = map(int, input().strip().split())
arr = []

def dfs():
    if len(arr) == M:
        print(' '.join(map(str, arr)))
        return

    for i in range(1, N+1):
        if i in arr:
            continue  # i가 arr에 존재하면 아래 코드들을 실행하지 않고 건너뜀
        arr.append(i)
        dfs()
        arr.pop()     # 다시 스택에서 빼줘야 이전 상황으로 돌아갈 수 있음

dfs()
```
