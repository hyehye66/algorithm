import sys, heapq
input = sys.stdin.readline
N = int(input().strip())
heap = []
for i in range(N):
    tup = tuple(map(int, input().strip().split()))
    for j in range(N):
        if len(heap) < N:
            heapq.heappush(heap, tup[j])
        else:
            if heap[0] < tup[j]:
                heapq.heappop(heap)
                heapq.heappush(heap, tup[j])

print(heapq.heappop(heap))
