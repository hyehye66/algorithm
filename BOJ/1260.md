```python
from collections import deque
N, M, V = map(int, input().strip().split())
arr = [[] for _ in range(N+1)]
visited = [0]*(N+1)
for i in range(M):
    a, b = map(int, input().strip().split())
    arr[a].append(b)
    arr[b].append(a)

for elm in arr:
    if elm:
        elm.sort()

def dfs(start):
    visited[start] = 1
    print(start, end=' ')

    for elm in arr[start]:
        if visited[elm] == 0:
            dfs(elm)

def bfs(start):
    visited2 = [0] * (N + 1)
    visited2[start] = 1
    print(start, end=' ')
    queue = deque([start])

    while queue:
        node = queue.popleft()
        for elm in arr[node]:
            if visited2[elm] == 0:
                visited2[elm] = 1
                print(elm, end=' ')
                queue.append(elm)


dfs(V)
print()
bfs(V)
```
