from collections import deque
def solution(maps):
    n = len(maps)
    m = len(maps[0])
    # 상우하좌
    dr = (-1,0,1,0)
    dc = (0,1,0,-1)
    
    def bfs():
        visited = [[0]*m for _ in range(n)]
        queue = deque([(0,0,1)])
        
        while queue:
            r, c, cnt = queue.popleft()
            if r == n-1 and c == m-1:
                return cnt            
            for i in range(4):
                nr, nc = r+dr[i], c+dc[i]
                if 0 <= nr < n and 0 <= nc< m:
                    if maps[nr][nc] == 1 and visited[nr][nc] == 0:                        
                        visited[nr][nc] = 1
                        queue.append((nr,nc, cnt+1))
        
        return -1
                        
    answer = bfs()
    return answer
