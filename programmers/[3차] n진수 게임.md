```python
def solution(n, t, m, p):
    answer = ''
    total = '0'
    string = 'ABCDEF'
    for i in range(1, t*m):
        temp = ''
        num = i
        while num:
            if 10 <= num%n <=15:
                temp += string[(num%n)%10]
            else:                
                temp += str(num%n)
            num //= n
        total += temp[::-1]
        if len(total) >= t*m:            
            break
    
    for i in range(t*m):
        if i%m == p-1:
            answer += total[i]

    return answer
```

[JavaScript]

```javascript
function solution(n, t, m, p) {
    let answer = '';
    let total = '0';
    string = 'ABCDEF';
    for (let i=1; i < t*m; i++) {
        temp = '';
        num = i;
        while (num > 0) {        
            if (10 <= num%n && num%n <= 15) {
                temp += string[(num%n)%10];
            } else {
                temp += num%n.toString();
            }
            num = parseInt(num/n);           
        }
        total += temp.split("").reverse().join("");
        if (total.length >= t*m) {
            break
        }
    }
    
    for (let i=0; i < t*m; i++) {
        if (i%m == p-1) {
            answer += total[i];
        }
    }
    
    return answer;
}
```
