```python

# 버블정렬 : 두 개의 자료를 서로 비교 후 더 작은 것을 앞으로 배치하는 것을 끝날 때까지 반복한다.
l = [8,6,4,2,1,3,5,7,9]
for j in range(len(l)):
    k = len(l)-j
    for i in range(1,k):
        if l[i-1] >= l[i]:
            temp = l[i-1]
            l[i-1] = l[i]
            l[i] = temp
print(l)
>>> [1,2,3,4,5,6,7,8,9]
```
