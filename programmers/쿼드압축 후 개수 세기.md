def solution(arr):    
    result = [0,0]
    def compress(a,b,l): 
        res = []
        flag = 1
        for i in range(a, a+l):
            for j in range(b, b+l):
                if arr[i][j] == arr[a][b]:
                    res.append(arr[i][j])
                else:
                    flag = 0
                    compress(a,b,l//2)
                    compress(a,b+l//2,l//2)
                    compress(a+l//2,b,l//2)
                    compress(a+l//2,b+l//2,l//2)
                    return

        if res.count(0) == len(res):
            result[0] += 1
        elif res.count(1) == len(res):
            result[1] += 1
        else:
            result[0] += res.count(0)
            result[1] += res.count(1)
        
        return
    
    compress(0,0,len(arr))
    
    return result
