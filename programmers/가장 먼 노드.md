1에서부터 각 노드에 도달하는 최단거리 중 가장 멀리 떨어진 노드를 구하는 문제이다. BFS로 각 노드까지 도달하는 거리를 구한 후 최대값에 해당하는 값의 개수를 헤아렸다.



[Python]

```python
from collections import deque
def solution(n, edge):
    answer = 0
    result = [0]*(n+1)
    max_val = 0
    arr = [[] for _ in range(n+1)]
    for e in edge:
        arr[e[0]].append(e[1])
        arr[e[1]].append(e[0])
    
    def bfs(x, c):
        nonlocal max_val
        
        visited[x] = 1
        queue = deque([(x,c)])
        
        while queue:
            node, cnt = queue.popleft()
            flag = 1    
            for next_node in arr[node]:
                if not visited[next_node]:
                    queue.append((next_node, cnt+1))
                    visited[next_node] = 1
                    flag = 0
            if flag:
                result[node] = cnt
                if cnt > max_val:
                    max_val = cnt                    

    visited = [0]*(n+1)
    bfs(1, 0)
    for i in range(1, n+1):
        if max_val == result[i]:
            answer += 1
    
    return answer
```
